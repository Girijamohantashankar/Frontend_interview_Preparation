Write a Promise syntax manually

const girija = new Promise((sipu, tapu) =>{
    let tiklu = true;

    if(tiklu){
        sipu("Promise was true");
    }else{
        tapu("promise was false");
    }
});

girija
  .then((res) => {
    console.log(res);
  }).catch((err) => {
    console.log(err);
  })
