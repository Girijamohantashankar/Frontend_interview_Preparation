2. what is currying?

currying = converting function with multiple arguments into nested functions 


function add(a){
    return function(b){
        return a + b;
    };
}
add(2)(3); //5


const multiply = (a) => (b) => (c) => a * b * c ;

console.log(multiply(2)(3)(4)); //24


3. Closures :-> Function remembers outer variables

function counter (){
    let count = 0;
    return {
        increment(){
            count++;
            console.log(count);
        },
        decrement(){
            count--;
            console.log(count);
        }
    };
}
const c = count();
c.increment(); //1
c.increment(); //2
c.decrement(); //1
