Write a function to count how many vowels are present in a string



const vowelCount = (str) => {
    let count = 0;
    const vowels = "aeiou";
    for (let i= 0; i< str.length; i++){
        if(vowels.includes(str[i].toLowerCase())){
            count++;
        };

    }
    return count;
}
console.log(vowelCount("React"));

🚀 Practical Question 8

Write a function to find the largest number in an array


const largestArray = (arr)  => {
    let max = arr[0];
    for (let i = 0; i< arr.length; i++){
        if(arr[i] > max){
            max = arr[i];
        }
    }
    return max;
}
console.log(largestArray([10, 50, 20, 80, 30]))

🚀 Practical Question 10

Write a function to count how many times each element appears in an array


const howmanytimeappears = (arr)  => {
    let obj = {};

    for(let i = 0; i<arr.length; i++){
        if(obj[arr[i]]){
            obj[arr[i]]++;
        }else{
            obj[arr[i]] = 1;
        }
    }
    return obj;

}
console.log(howmanytimeappears([1, 2, 2, 3, 3, 3]));
