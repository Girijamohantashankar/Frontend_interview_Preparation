and:- throttling ensures a function runs at fixed intervals


scolling page 

wothout throttle --> 1000 calls

function throttle (fn , limit){
    let flag = true;

    return function (){
        if (flag){
            fn();
            flag = false;
            setTimeOut(() => {
                flag = true;
            }, limit);
        }
    };
}

function handleScroll() {
    console.log("Scrolling...");
}
window.addEventListener("scroll",throttle(handleScroll, 1000));
